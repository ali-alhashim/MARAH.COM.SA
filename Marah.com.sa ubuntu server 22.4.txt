### update server

sudo apt update && sudo apt upgrade

### enable firewall

sudo ufw enafble

### install vsftpd

sudo apt install vsftpd

### add ftp user

sudo adduser aliftp

### allow port 20 & 21 for using ftp

sudo ufw allow 20/tcp
sudo ufw allow 80
sudo ufw allow 443
sudo ufw allow 21/tcp
sudo ufw allow 30000:31000/tcp
sudo ufw allow ftp
sudo ufw allow 'Nginx Full'
sudo ufw allow 22
### use root user 
sudo -i
### go home and go to ailftp and create folder for ftp root folder 
mkdir MARAH

## chown

sudo chown aliftp:nogroup MARAH




### edit vsftpd conf file

sudo nano /etc/vsftpd.conf

###go to last and add the following :

local_enable=YES
write_enable=YES
pasv_enable=YES
pasv_min_port=30000
pasv_max_port=31000
chroot_local_user=YES
user_sub_token=$USER
local_root=/home/aliftp/MARAH
userlist_enable=YES
userlist_file=/etc/vsftpd.allowed_users
userlist_deny=NO
allow_writeable_chroot=YES
### save changes and restart vsftpd
sudo service vsftpd restart

### create file and add user 

sudo nano /etc/vsftpd.allowed_users
